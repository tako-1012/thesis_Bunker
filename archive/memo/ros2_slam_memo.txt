############################################################
## ROS 2 SLAM (単体ロボット) 実行手順まとめ
############################################################

# 目的: 
# TurtleBot3ロボットを使い、標準のGazeboワールド（ポールのあるワールド）の
# 地図を作成する（SLAM）。

# 必要なもの: 
# 合計4つのターミナル

#-----------------------------------------------------------
# 手順1：シミュレーターの起動 (ターミナル1)
#-----------------------------------------------------------

# 1. ターミナルを開く
# 2. 以下のコマンドを順番に実行する

export TURTLEBOT3_MODEL=burger
ros2 launch turtlebot3_gazebo turtlebot3_world.launch.py


#-----------------------------------------------------------
# 手順2：SLAMノードの起動 (ターミナル2)
#-----------------------------------------------------------

# 1. 新しいターミナルを開く
# 2. 以下のコマンドを実行する

ros2 launch slam_toolbox online_async_launch.py


#-----------------------------------------------------------
# 手順3：RViz2（可視化ツール）の起動 (ターミナル3)
#-----------------------------------------------------------

# 1. 新しいターミナルを開く
# 2. 以下のコマンドを実行する

rviz2


#-----------------------------------------------------------
# 手順4：キーボード操作ノードの起動 (ターミナル4)
#-----------------------------------------------------------

# 1. 新しいターミナルを開く
# 2. 以下のコマンドを順番に実行する

export TURTLEBOT3_MODEL=burger
ros2 run turtlebot3_teleop teleop_keyboard


#-----------------------------------------------------------
# RViz2の初期設定
#-----------------------------------------------------------

# 初回起動時のみ、起動したRViz2で以下の手動設定が必要。

# 1. 左下の [Add] ボタンをクリック。
# 2. [By topic] タブから [/map] を選択して追加。
# 3. 再度 [Add] ボタンをクリック。
# 4. [By topic] タブから [/scan] を選択して追加。
# 5. 再度 [Add] ボタンをクリック。
# 6. [By display type] タブから [TF] を選択して追加。


#-----------------------------------------------------------
# 地図作成の実行
#-----------------------------------------------------------

# 1. ターミナル4 (キーボード操作用) をクリックして選択。
# 2. [w][a][d][x] キーでロボットを操作する。
# 3. RViz2のウィンドウで、リアルタイムに地図が作成されるのを確認する。
